CREATE TABLE school(
    school_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    name varchar (128)
);

CREATE TABLE teachers( 
    id_teacher INT NOT NULL AUTO_INCREMENT PRIMARY KEY, 
    name varchar(128) NOT NULL, school_id INT NOT NULL, 
    CONSTRAINT FOREIGN KEY(school_id) REFERENCES school (school_id) ON DELETE CASCADE ON UPDATE CASCADE );


CREATE TABLE comments (
    id_comentario INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    comment VARCHAR(512) NOT NULL,
    por INT NOT NULL,
    para INT NOT NULL,
    fecha DATETIME NOT NULL,

    CONSTRAINT FOREIGN KEY (por)
    	REFERENCES users (user_id)
    	ON DELETE CASCADE ON UPDATE CASCADE,
    CONSTRAINT FOREIGN KEY (para)
    	REFERENCES teachers (id_teacher)
    	ON DELETE CASCADE ON UPDATE CASCADE
);
